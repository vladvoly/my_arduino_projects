#include <Arduino.h>
#include <QN8027.h>
#include <Wire.h>

QN8027 fm;
float frequency = 88.5;

void setup() {
  Serial.begin(9600);
  fm.begin(frequency);
  fm.setTxPower(70);
  fm.enable(true);
}

void loop() {

  if (Serial.available() > 0) {
    frequency = Serial.parseFloat();
    if (frequency > 0) {
      Serial.print("I received: ");
      Serial.println(frequency, 1);
      fm.reset();
      fm.begin(frequency);
      fm.enable(true);
    }
  }
}
